javascript: (function() {
	setInterval(function() {
		Game.goldenCookie.life > 0 && !Game.goldenCookie.wrath && Game.goldenCookie.click();
	}, 2000);
	setInterval(function() {
		Game.ClickCookie();
	}, 1000 / 250);
	setInterval(function() {
		var umin = Infinity;
		for (var i in Game.UpgradesInStore) {
			if (Game.UpgradesInStore[i].name.indexOf('Revoke')!=0) {
				var u = Game.UpgradesInStore[i];
				umin = Math.min(umin, u.basePrice);
				u.buy();
			}
		}
		var best = -1,
			best_cost = Infinity;
		for (var i = Game.ObjectsById.length - 1; i >= 0; --i) {
			var o = Game.ObjectsById[i],
				p = o.basePrice * Math.pow(Game.priceIncrease, o.amount + 2),
				c = o.price * (o.amount += 1) / o.cps();
				o.amount -= 1; o.cps();
			Game.cookies >= 0.9 * o.price && p < umin && c < best_cost && (best = i) && (best_cost = c);
		}
		best != -1 && Game.ObjectsById[best].buy();
	}, 1000);
})();
